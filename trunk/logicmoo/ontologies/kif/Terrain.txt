(subrelation bottom superficialPart)
(domain bottom 1 Region)
(domain bottom 2 Region)
(documentation bottom 
"(bottom ?Bottom ?Region) holds if ?Bottom is the lowest or deepest part of ?Region")

(=>
	(and
		(bottom ?Bottom ?Object)
		(part ?Part ?Object)
		(not (connected ?Part ?Bottom)))
	(above ?Part ?Bottom))

(subrelation top superficialPart)
(domain top 1 Region)
(domain top 2 Region)
(documentation top
"(top ?Top ?Region) means that ?Top is the highest part or point of ?Region")

(=>
	(and
		(top ?Top ?Object)
		(part ?Part ?Object)
		(not (connected ?Part ?Top)))
	(below ?Part ?Top))

(<=>
	(exists (?Top ?Bottom)
		(and
			(top ?Top ?Object)
			(bottom ?Bottom ?Object)
			(distance ?Top ?Bottom ?Height)))
	(height ?Object ?Height))


(subrelation rim superficialPart)
(domain rim 1 SelfConnectedObject)
(domain rim 2 SelfConnectedObject)
(documentation rim
"(rim ?Rim ?Object) means that ?Rim is the outer, often curved or circular, edge of ?Object")



(subclass Freezing NaturalProcess)
(documentation Freezing
"A process of changing from the liquid to the solid state by loss of heat")

(=>
	(and
		(instance ?Freezing Freezing)
		(patient ?Freezing ?Substance))
	(and
		(holdsDuring (ImmediatePastFn (WhenFn ?Freezing)) (attribute ?Substance Liquid))
		(holdsDuring (ImmediateFutureFn (WhenFn ?Freezing)) (attribute ?Substance Solid))))
		

(subclass Melting NaturalProcess)
(documentation Melting
"A process of changing from a solid to a liquid state especially by the application of heat.")

(=>
	(and
		(instance ?Melting Melting)
		(patient ?Melting ?Substance))
	(and
		(holdsDuring (ImmediatePastFn (WhenFn ?Melting)) (attribute ?Substance Solid))
		(holdsDuring (ImmediateFutureFn (WhenFn ?Freezing)) (attribute ?Substance Liquid))))


(subrelation altitude height)
(domain altitude 1 Region)
(domain altitude 2 ConstantQuantity)
(documentation altitude
"The height above sea level")

(=>
	(and
		(altitude ?Region ?Elevation)
		(top ?Top ?Region))
	(distance ?Top SeaLevel ?Elevation))


(instance SeaLevel SelfConnectedObject)
(documentation SeaLevel
"The level of the surface of the sea relative to the land, halfway between high and low tide, used as a  standard in calculating elevation")

(instance Flat Attribute)
(documentation Flat
"Having a horizontal surface without a slope, tilt, or curvature.")

(=>
	(and
		(instance ?Region Region)
		(attribute ?Region Flat)
		(part ?Part1 ?Region)
		(part ?Part2 ?Region)
		(altitude ?Part1 ?Height1)
		(altitude ?Part2 ?Height2))
	(equal ?Height1 ?Height2))

(=>
	(and
		(instance ?Region Region)
		(attribute ?Region Flat))
	(not
		(exists (?Slope)
			(and
				(instance ?Slope Slope)
				(part ?Slope ?Region)))))

(subclass WaterRegion Region)
(documentation WaterRegion
"A body of water")


(=>
	(instance ?WaterRegion WaterRegion)
	(exists (?Hole ?Water)
		(and
			(instance ?Hole Hole)
			(instance ?Water Water)
			(hole ?Hole ?WaterRegion)
			(properlyFills ?Water ?Hole)))) 
			

(subclass EarthRegion Region)
(documentation EarthRegion
"An area of solid ground")

(=>
	(instance ?Region EarthRegion)
	(attribute ?Region Solid))


(=>
	(and
		(instance ?Land EarthRegion)
		(attribute ?Land Dry)
		(soilType ?Land ?SoilType))
	(not (subclass ?SoilType FertileSoil)))

(subclass Slope Region)
(documentation Slope
"A surface which lies at angle to the horizontal so that some points on it are higher than others")

(=>
	(and
		(instance ?Slope Slope)
		(part ?Part1 ?Slope)
		(part ?Part2 ?Slope)
		(not (overlapsSpatially ?Part1 ?Part2))
		(altitude ?Part1 ?Height1)
		(altitude ?Part2 ?Height2))
	(not (equal ?Height1 ?Height2)))

(documentation Veld
"Grassy, undulating plateaus of South Africa and Zimbabwe, ranging in elevation from 150 to 1,830 m.")

(=>
	(and
		(instance ?Veld Veld)
		(altitude ?Veld ?Altitude))
	(and
	 	(greaterThan ?Altitude (MeasureFn 150 Meter))
		(greaterThan (MeasureFn 1830 Meter) ?Altitude)))

(subclass Flight Motion)
(documentation Flight
"The process or act of moving through the air or through space")

(=>
	(and
		(instance ?Flight Flight)
		(path ?Flight ?Path))
	(exists (?Air)
		(and
			(instance ?Air Air)
			(part ?Path ?Air))))

(subclass Landing Motion)
(documentation Landing
"The act or process of coming to land after a flight")

(=>
	(and
		(instance ?Landing Landing)
		(instrument ?Landing ?Plane)
		(destination ?Landing ?Region))
	(exists (?Flight ?Air)
		(and
			(instance ?Flight Flight)
			(instance ?Air Air)
			(holdsDuring (ImmediatePastFn (WhenFn ?Landing)) (located ?Plane ?Air))
			(holdsDuring (ImmediateFutureFn (WhenFn ?Landing)) (located ?Plane ?Region)))))
